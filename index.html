const fs = require('fs'); const path = require('path'); // Arquivo de estado em /tmp (área de escrita das funções) const FILE = path.join('/tmp', 'regua-valor-state.json'); exports.handler = async function(event) { const headers = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'Content-Type', 'Content-Type': 'application/json' }; if (event.httpMethod === 'OPTIONS') { return { statusCode: 200, headers, body: '' }; } if (event.httpMethod === 'GET') { try { if (fs.existsSync(FILE)) { const data = fs.readFileSync(FILE, 'utf8'); return { statusCode: 200, headers, body: data }; } return { statusCode: 200, headers, body: '{}' }; } catch (e) { return { statusCode: 200, headers, body: '{}' }; } } if (event.httpMethod === 'POST') { try { fs.writeFileSync(FILE, event.body || '{}', 'utf8'); return { statusCode: 200, headers, body: '{"ok":true}' }; } catch (e) { return { statusCode: 500, headers, body: '{"ok":false}' }; } } return { statusCode: 405, headers, body: '{"error":"method not allowed"}' }; };
